1. Rektangelmetoden¶

def f(x):
    return x + 2

a = 1 # x_0 = 1
b = 4 # x_N = 4

# Hva er integralet av f(x) mellom 1 og 4?

A = 0
N = 3
dx = (b - a)/N # dx = (4-1)/3 = 1

for i in range(N):
    A = A + f(a + i*dx)*dx

print("Integralet av f(x) mellom", a, "og", b, "er:", A)
print(f"Integralet av f(x) mellom {a} og {b} er {A}.")

def f(x):
    return x + 2

def trapesmetoden(f, a, b, N):
    dx = (b-a)/N
    A = (f(a) + f(b))/2
    for i in range(1,N):
        A = A + f(a + i*dx)
    return A*dx

itsatrap = trapesmetoden(f, 1, 4, 1)
print(itsatrap)

13.5
