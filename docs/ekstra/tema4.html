from pylab import *

funksjonsverdier = []
tid = []

def g(t):
    return sqrt(t) + 2*log10(t)

for t in range(1,100):
    funksjonsverdier.append(g(t))
    tid.append(t)
    
plot(tid,funksjonsverdier)
show()
